            <div n="6" type="sthāna">
                <head>[uttaratantram]</head>

                <div n="1" type="adhyāya">
                    <head>prathamo 'dhyāyaḥ | </head>
                    <l xml:id="SS.6.1.1"> athāta aupadravikam adhyāyaṃ vyākhyāsyāmaḥ ||</l>
                    <l xml:id="SS.6.1.2"> yathovāca bhagavān dhanvantariḥ ||</l>
                    <l xml:id="SS.6.1.3"> adhyāyānāṃ śate viṃśe yad uktam asakṛn mayā |
                        <caesura/>
                         vakṣyāmi bahudhā samyag uttare 'rthān imān iti || </l>
                    <l xml:id="SS.6.1.4"> idānīṃ tat pravakṣyāmi tantram uttaram uttamam |
                        <caesura/>
                         nikhilenopadiśyante yatra rogāḥ pṛthagvidhāḥ || </l>
                    <l xml:id="SS.6.1.5"> śālākyatantrābhihitā videhādhipakīrtitāḥ |
                        <caesura/>
                         ye ca vistarato dṛṣṭāḥ kumārābādhahetavaḥ || </l>

                    <l xml:id="SS.6.1.6"> ṣaṭsu kāyacikitsāsu ye coktāḥ paramarṣibhiḥ |
                        <caesura/>
                         upasargādayo rogā ye cāpy āgantavaḥ smṛtāḥ || </l>

                    <l xml:id="SS.6.1.7"> triṣaṣṭī rasasaṃsargāḥ svasthavṛttaṃ tathaiva ca |
                        <caesura/>
                         yuktārthā yuktayaś caiva doṣabhedās tathaiva ca || </l>

                    <l xml:id="SS.6.1.8"> yatroktā vividhā arthā rogasādhanahetavaḥ |
                        <caesura/>
                         mahatas tasya tantrasya durgādhasyāmbudher iva || </l>

                    <l xml:id="SS.6.1.9"> ādāv evottamāṅgasthān rogān abhidadhāmy aham |
                        <caesura/>
                         saṅkhyayā lakṣaṇaiś cāpi sādhyāsādhyakrameṇa ca || </l>

                    <l xml:id="SS.6.1.10"> vidyāddvayaṅgulabāhulyaṃ svāṅguṣṭhodarasaṃmitam |
                        <caesura/>
                         dvyaṅgulaṃ sarvataḥ sārdhaṃ bhiṣaṅnayanabudbudam || </l>

                    <l xml:id="SS.6.1.11"> suvṛttaṃ gostanākāraṃ sarvabhūtaguṇodbhavam |
                        <caesura/>
                         palaṃ bhuvo 'gnito raktaṃ vātāt kṛṣṇaṃ sitaṃ jalāt || </l>
                    <l xml:id="SS.6.1.12"> ākāśād aśrumārgāś ca jāyante netrabudbude |
                        <caesura/>
                         dṛṣṭiṃ cātra tathā vakṣye yathā brūyād viśāradaḥ || </l>
                    <l xml:id="SS.6.1.13"> netrāyāmatribhāgaṃ tu kṛṣṇamaṇḍalamucyate |
                        <caesura/>
                         kṛṣṇāt saptamam icchanti dṛṣṭiṃ dṛṣṭiviśāradāḥ || </l>
                    <l xml:id="SS.6.1.14"> maṇḍalāni ca sandhīṃś ca paṭalāni ca locane |
                        <caesura/>
                         yathākramaṃ vijānīyāt pañca ṣaṭ ca ṣaḍeva ca || </l>
                    <l xml:id="SS.6.1.15"> pakṣmavartmaśvetakṛṣṇadṛṣṭīnāṃ maṇḍalāni tu |
                        <caesura/>
                         anupūrvaṃ tu te madhyāś catvāro 'ntyā yathottaram || </l>
                    <l xml:id="SS.6.1.16"> pakṣmavartmagataḥ sandhir vartmaśuklagato 'paraḥ |
                        <caesura/>
                         śuklakṛṣṇagatas tv anyaḥ kṛṣṇadṛṣṭigato 'paraḥ |
                        <caesura/>
                         tataḥ kanīnakagataḥ ṣaṣṭhaś cāpāṅgagaḥ smṛtaḥ || </l>
                    <l xml:id="SS.6.1.17"> dve vartmapaṭale vidyāc catvāry anyāni cākṣiṇi |
                        <caesura/>
                         jāyate timiraṃ yeṣu vyādhiḥ paramadāruṇaḥ || </l>
                    <l xml:id="SS.6.1.18"> tejo jalāśritaṃ bāhyaṃ teṣv anyat piśitāśritam |
                        <caesura/>
                         medas tṛtīyaṃ paṭalam āśritaṃ tvasthi cāparam || </l>
                    <l xml:id="SS.6.1.19"> pañcamāṃśasamaṃ dṛṣṭes teṣāṃ bāhulyam iṣyate |
                        <caesura/>
                         sirāṇāṃ kaṇḍarāṇāṃ ca medasaḥ kālakasya ca || </l>
                    <l xml:id="SS.6.1.20"> guṇāḥ kālāt paraḥ śleṣmā bandhane 'kṣṇoḥ sirāyutaḥ |
                        <caesura/>
                         sirānusāribhir doṣair viguṇair ūrdhvam āgataiḥ || </l>
                    <l xml:id="SS.6.1.21"> jāyante netrabhāgeṣu rogāḥ paramadāruṇāḥ |
                        <caesura/>
                         tatrāvilaṃ sasaṃrambham aśrukaṇḍūpadehavat || </l>
                    <l xml:id="SS.6.1.22"> gurūṣātodarāgādyair juṣṭaṃ cāvyaktalakṣaṇaiḥ |
                        <caesura/>
                         saśūlaṃ vartmakoṣeṣu śūkapūrṇābham eva ca || </l>
                    <l xml:id="SS.6.1.23"> vihanyamānaṃ rūpe vā kriyāsv akṣi yathā purā |
                        <caesura/>
                         dṛṣṭvaiva dhīmān budhyeta doṣeṇādhiṣṭhitaṃ tu tat || </l>
                    <l xml:id="SS.6.1.24"> tatra saṃbhavam āsādya yathādoṣaṃ bhiṣagjitam |
                        <caesura/>
                         vidadhyān netrajā rogā balavantaḥ syur anyathā || </l>
                    <l xml:id="SS.6.1.25"> saṅkṣepataḥ kriyāyogo nidānaparivarjanam |
                        <caesura/>
                         vātādīnāṃ pratīghātaḥ prokto vistarataḥ punaḥ || </l>
                    <l xml:id="SS.6.1.26"> uṣṇābhitaptasya jalapraveśād dūrekṣaṇāt svapnaviparyayāc
                        ca |
                        <caesura/>
                         prasaktasaṃrodanakopaśokakleśābhighātād atimaithunāc ca || </l>
                    <l xml:id="SS.6.1.27"> śuktāranālāmlakulatthamāṣaniṣevaṇād vegavinigrahāc ca |
                        <caesura/>
                         svedād atho dhūmaniṣevaṇāc ca charder vighātād vamanātiyogāt |
                        <caesura/>
                         bāṣpagrahāt sūkṣmanirīkṣaṇāc ca netre vikārān janayanti doṣāḥ || </l>
                    <l xml:id="SS.6.1.28"> vātād daśa tathā pittāt kaphāc caiva trayodaśa |
                        <caesura/>
                         raktāt ṣoḍaśa vijñeyāḥ sarvajāḥ pañcaviṃśatiḥ || </l>
                    <l xml:id="SS.6.1.29"> tathā bāhyau punar dvau ca rogāḥ ṣaṭsaptatiḥ smṛtāḥ |
                        <caesura/>
                         hatādhimantho nimiṣo dṛṣṭir gambhīrikā ca yā || </l>
                    <l xml:id="SS.6.1.30"> yac ca vātahataṃ vartma na te sidhyanti vātajāḥ |
                        <caesura/>
                         yāpyo 'tha tanmayaḥ kācaḥ sādhyāḥ syuḥ sānyamārutāḥ || </l>
                    <l xml:id="SS.6.1.31"> śuṣkākṣipākādhīmanthasyandamārutaparyayāḥ |
                        <caesura/>
                         asādhyo hrasvajāḍyo yo jalasrāvaś ca paittikaḥ || </l>
                    <l xml:id="SS.6.1.32"> parimlāyī ca nīlaś ca yāpyaḥ kāco 'tha tanmayaḥ |
                        <caesura/>
                         abhiṣyando 'dhimantho 'mlādhyuṣitaṃ śuktikā ca yā || </l>
                    <l xml:id="SS.6.1.33"> dṛṣṭiḥ pittavidagdhā ca dhūmadarśī ca sidhyati |
                        <caesura/>
                         asādhyaḥ kaphajaḥ srāvo yāpyaḥ kācaś ca tanmayaḥ || </l>
                    <l xml:id="SS.6.1.34"> abhiṣyando 'dhimanthaś ca balāsagrathitaṃ ca yat |
                        <caesura/>
                         dṛṣṭiḥ śleṣmavidagdhā ca pothakyo lagaṇaś ca yaḥ || </l>
                    <l xml:id="SS.6.1.35">krimigranthipariklinnavartmaśuklārmapiṣṭakāḥ |
                        <caesura/>
                         śleṣmopanāhaḥ sādhyās tu kathitāḥ śleṣmajeṣu tu || </l>
                    <l xml:id="SS.6.1.36"> raktasrāvo 'jakājātaṃ śoṇitārśovraṇānvitam |
                        <caesura/>
                         śukraṃ na sādhyaṃ kācaś ca yāpyastajjaḥ prakīrtitaḥ || </l>
                    <l xml:id="SS.6.1.37"> manthasyandau kliṣṭavartma harṣotpātau tathaiva ca |
                        <caesura/>
                         sirājātāñjanākhyā ca sirājālaṃ ca yat smṛtam || </l>
                    <l xml:id="SS.6.1.38"> parvaṇyathāvraṇaṃ śukraṃ śoṇitārmārjunaś ca yaḥ |
                        <caesura/>
                         ete sādhyā vikāreṣu raktajeṣu bhavanti hi || </l>
                    <l xml:id="SS.6.1.39"> pūyāsrāvo nākulāndhyamakṣipākātyayo 'lajī |
                        <caesura/>
                         asādhyāḥ sarvajā yāpyaḥ kācaḥ kopaś ca pakṣmaṇaḥ || </l>
                    <l xml:id="SS.6.1.40"> vartmāvabandho yo vyādhiḥ sirāsu piḍakā ca yā |
                        <caesura/>
                         prastāryarmādhimāṃsārma snāyvarmotsaṅginī ca yā || </l>
                    <l xml:id="SS.6.1.41"> pūyālasaś cārbudaṃ ca śyāvakardamavartmanī |
                        <caesura/>
                         tathārśovartma śuṣkārśaḥ śarkarāvartma yac ca vai || </l>
                    <l xml:id="SS.6.1.42"> saśophaś cāpy aśophaś ca pāko bahalavartma ca |
                        <caesura/>
                         aklinnavartma kumbhīkā bisavartma ca sidhyati || </l>
                    <l xml:id="SS.6.1.43"> sanimitto 'nimittaś ca dvāv asādhyau tu bāhyajau |
                        <caesura/>
                         ṣaṭsaptatir vikārāṇām eṣā saṃgrahakīrtitā || </l>
                    <l xml:id="SS.6.1.44"> nava sandhyāśrayās teṣu vartmajās tv ekaviṃśatiḥ |
                        <caesura/>
                         śuklabhāge daśaikaś ca catvāraḥ kṛṣṇabhāgajāḥ || </l>
                    <l xml:id="SS.6.1.45"> sarvāśrayāḥ saptadaśa dṛṣṭijā dvādaśaiva tu |
                        <caesura/>
                         bāhyajau dvau samākhyātau rogau paramadāruṇau |
                        <caesura/>
                         bhūya etān pravakṣyāmi saṅkhyārūpacikitsitaiḥ ||</l>
                    <p ana="trailer">iti suśrutasaṃhitāyām uttaratantrāntargate śālākyatantre
                        aupadraviko nāma prathamo 'dhyāyaḥ ||1 ||</p>
                </div>


                <div n="2" type="adhyāya">

                    <head>dvitīyo 'dhyāyaḥ | </head>


                    <l xml:id="SS.6.2.1"> athātaḥ sandhigatarogavijñānīyam adhyāyaṃ vyākhyāsyāmaḥ || </l>

                    <l xml:id="SS.6.2.2"> yathovāca bhagavān dhanvantariḥ || </l>

                    <l xml:id="SS.6.2.2-i"/>
                    <l xml:id="SS.6.2.2-ii"/>


                    <l xml:id="SS.6.2.3"> pūyālasaḥ sopanāhaḥ srāvāḥ parvaṇikālajī |
                        <caesura/>
                         krimigranthiś ca vijñeyā rogāḥ sandhigatā nava || </l>

                    <l xml:id="SS.6.2.4"> pakvaḥ śophaḥ sandhijaḥ saṃsravedyaḥ sāndraṃ pūyaṃ pūti
                        pūyālasaḥ saḥ |
                        <caesura/>
                         granthirnālpo dṛṣṭisandhāvapākaḥ kaṇḍūprāyo nīrujastūpanāhaḥ || </l>

                    <l xml:id="SS.6.2.5"> gatvā sandhīnaśrumārgeṇa doṣāḥ kuryuḥ srāvān rugvihīnān
                        kanīnāt |
                        <caesura/>
                         tān vai srāvān netranāḍīm athaike tasyā liṅgaṃ kīrtayiṣye caturdhā || </l>

                    <l xml:id="SS.6.2.6"> pākaḥ sandhau saṃsravedyaś ca pūyaṃ pūyāsrāvo naikarūpaḥ
                        pradiṣṭaḥ |
                        <caesura/>
                         śvetaṃ sāndraṃ picchilaṃ saṃsravedyaḥ śleṣmāsrāvo nīrujaḥ sa pradiṣṭaḥ || </l>

                    <l xml:id="SS.6.2.7"> raktāsrāvaḥ śoṇitotthaḥ saraktamuṣṇaṃ nālpaṃ
                        saṃsravennātisāndram |
                        <caesura/>
                         pītābhāsaṃ nīlamuṣṇaṃ jalābhaṃ pittāsrāvaḥ saṃsravet sandhimadhyāt || </l>


                    <l xml:id="SS.6.2.8"> tāmrā tanvī dāhaśūlopapannā raktājjñeyā parvaṇī vṛttaśophā
                        |
                        <caesura/>
                         jātā sandhau kṛṣṇaśukle 'lajī syāt tasminn eva khyāpitā pūrvaliṅgaiḥ || </l>


                    <l xml:id="SS.6.2.9"> krimigranthirvartmanaḥ pakṣmaṇaś ca kaṇḍūṃ kuryuḥ krimayaḥ
                        sandhijātāḥ |
                        <caesura/>
                         nānārūpā vartmaśuklasya sandhau caranto 'ntarnayanaṃ dūṣayanti ||</l>


                    <p ana="trailer">iti suśrutasaṃhitāyām uttaratantre sandhigatarogavijñānīyo nāma
                        dvitīyo 'dhyāyaḥ ||2 ||</p>
                </div>



                <div n="3" type="adhyāya">
                    <head>tṛtīyo 'dhyāyaḥ | </head>
                    <l xml:id="SS.6.3.1"> athāto vartmagatarogavijñānīyam adhyāyaṃ vyākhyāsyāmaḥ
                        ||</l>

                    <l xml:id="SS.6.3.2"> yathovāca bhagavān dhanvantariḥ ||</l>

                    <l xml:id="SS.6.3.3"> pṛthagdoṣāḥ samastā vā yadā vartmavyapāśrayāḥ |
                        <caesura/>
                         sirā vyāpyāvatiṣṭhante vartmasvadhikamūrcchitāḥ || </l>


                    <l xml:id="SS.6.3.4"> vivardhya māṃsaṃ raktaṃ ca tadā vartmavyapāśrayān |
                        <caesura/>
                         vikārāñjanayanty āśu nāmatas tān nibodhata || </l>


                    <l xml:id="SS.6.3.5"> utsaṅginyatha kumbhīkā pothakyo vartmaśarkarā |
                        <caesura/>
                         tathārśovartma śuṣkārśastathaivāñjananāmikā || </l>


                    <l xml:id="SS.6.3.6"> bahalaṃ vartma yaccāpi vyādhirvartmāvabandhakaḥ |
                        <caesura/>
                         kliṣṭakardamavartmākhyau śyāvavartma tathaiva ca || </l>


                    <l xml:id="SS.6.3.7"> praklinnamapariklinnaṃ vartma vātahataṃ tu yat |
                        <caesura/>
                         arbudaṃ nimiṣaś cāpi śoṇitārśaś ca yat smṛtam || </l>


                    <l xml:id="SS.6.3.8"> lagaṇo biśanāmā ca pakṣmakopas tathaiva ca |
                        <caesura/>
                         ekaviṃśatirityete vikārā vartmasaṃśrayāḥ || </l>


                    <l xml:id="SS.6.3.9"> nāmabhis te samuddiṣṭā lakṣaṇaistān pracakṣmahe |
                        <caesura/>
                         abhyantaramukhī bāhyotsaṅge 'dho vartmanaś ca yā || </l>


                    <l xml:id="SS.6.3.10"> vijñeyotsaṅginī nāma tadrūpapiḍakācitā |
                        <caesura/>
                         kumbhīkabījapratimāḥ piḍakā yās tu vartmajāḥ || </l>


                    <l xml:id="SS.6.3.11"> ādhmāpayanti bhinnā yāḥ kumbhīkapiḍakās tu tāḥ |
                        <caesura/>
                         srāviṇyaḥ kaṇḍurā gurvyo raktasarṣapasannibhāḥ |
                        <caesura/>
                         piḍakāś ca rujāvatyaḥ pothakya iti saṃjñitāḥ || </l>


                    <l xml:id="SS.6.3.12"> piḍakābhiḥ susūkṣmābhir ghanābhir abhisaṃvṛtā |
                        <caesura/>
                         piḍakā yā kharā sthūlā sā jñeyā vartmaśarkarā || </l>


                    <l xml:id="SS.6.3.13"> ervārubījapratimāḥ piḍakā mandavedanāḥ |
                        <caesura/>
                         sūkṣmāḥ kharāś ca vartmasthāstadarśovartma kīrtyate || </l>


                    <l xml:id="SS.6.3.14"> dīrgho 'ṅkuraḥ kharaḥ stabdho dāruṇo vartmasaṃbhavaḥ |
                        <caesura/>
                         vyādhireṣa samākhyātaḥ śuṣkārśa iti saṃjñitaḥ || </l>
