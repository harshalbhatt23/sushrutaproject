                <div n="1" type="adhyāya">
                    <head>prathamo 'dhyāyaḥ |</head>
                    <l xml:id="SS.3.1.1">
                        <label>1931 ed. SS.3.1.1</label>
                        <caesura/>
                         athātaḥ sarvabhūtacintāśārīraṃ vyākhyāsyāmaḥ || </l>

                    <l xml:id="SS.3.1.2">
                        <label>1931 ed. SS.3.1.2</label>
                        <caesura/>
                         yathovāca bhagavān dhanvantariḥ || </l>

                    <l xml:id="SS.3.1.3">
                        <label>1931 ed. SS.3.1.3</label>
                        <caesura/>
                         sarvabhūtānāṃ kāraṇam akāraṇaṃ sattvarajastamo<pc> </pc>lakṣaṇam
                        aṣṭarūpam akhilasya jagataḥ saṃbhavahetur avyaktaṃ nāma | tad ekaṃ
                        bahūnāṃ kṣetrajñānām adhiṣṭhānaṃ samudra ivaudakānāṃ bhāvānām || </l>

                    <l xml:id="SS.3.1.4">
                        <label>1931 ed. SS.3.1.4</label>
                        <caesura/>
                         tasmād avyaktān mahān utpadyate talliṅga eva; talliṅgāc ca
                        mahatas tallakṣaṇa evāhaṅkāra utpadyate, sa trividho vaikārikas
                        taijaso bhūtādir iti; tatra vaikārikād ahaṅkārāt taijasasahāyāt
                        tallakṣaṇāny enaikādaśendriyāṇy utpadyante, tad yathā ---
                            śrotra<pc> </pc>tvakcakṣurjihvā<pc> </pc>ghrāṇa<pc
                            > </pc>vāgghastopastha<pc> </pc>pāyu<pc> </pc>pāda<pc
                        > </pc>manāṃsīti, tatra pūrvāṇi pañca buddhīndriyāṇi, itarāṇi
                        pañca karmendriyāṇi, ubhayātmakaṃ manaḥ; bhūtāder api
                        taijasasahāyāt tallakṣaṇāny eva pañcatanmātrāṇy utpadyante, tad
                        yathā --- śabdatanmātraṃ, sparśatanmātraṃ, rūpatanmātraṃ,
                        rasatanmātraṃ, gandhatanmātram iti; teṣāṃ viśeṣāṃ viśeṣāḥ
                        śabdasparśarūparasagandhāḥ; tebhyo bhūtāni vyomānilānalajalorvyaḥ;
                        evam eṣā tattvacaturviṃśatir vyākhyātā || </l>

                    <l xml:id="SS.3.1.5">
                        <label>1931 ed. SS.3.1.5</label>
                        <caesura/>
                         tatra, buddhīndriyāṇāṃ śabdādayo viṣayāḥ; karmendriyāṇāṃ
                        yathāsaṅkhyaṃ vacanādānānandavisargaviharaṇāni || </l>

                    <l xml:id="SS.3.1.6">
                        <label>1931 ed. SS.3.1.6</label>
                        <caesura/>
                         avyaktaṃ mahān ahaṅkāraḥ pañca tanmātrāṇi cety aṣṭau prakṛtayaḥ;
                        śeṣāḥ ṣoḍaśa vikārāḥ || </l>

                    <l xml:id="SS.3.1.7">
                        <label>1931 ed. SS.3.1.7</label>
                        <caesura/>
                         svaḥ svaś caiṣāṃ viṣayo 'dhibhūtaṃ; svayam adhyātmaṃ;
                        adhidaivataṃ tu --- buddher brahmā, ahaṅkārasyeśvaraḥ, manasaś
                        candramāḥ, diśaḥ śrotrasya, tvaco vāyuḥ, sūryaś cakṣuṣaḥ,
                        rasanasyāpaḥ, pṛthivī ghrāṇasya, vāco 'gniḥ, hastayor indraḥ,
                        pādayor viṣṇuḥ, pāyor mitraḥ, prajāpatir upasthasyeti || </l>

                    <l xml:id="SS.3.1.8">
                        <label>1931 ed. SS.3.1.8</label>
                        <caesura/>
                         tatra sarva evācetana eṣa vargaḥ, puruṣaḥ pañcaviṃśatitamaḥ
                        kāryakāraṇasaṃyuktaś cetayitā bhavati | saty apy acaitanye
                        pradhānasya puruṣakaivalyārthaṃ pravṛttim upadiśanti kṣīrādīṃś
                        cātra hetūn udāharanti || </l>

                    <l xml:id="SS.3.1.9">
                        <label>1931 ed. SS.3.1.9</label>
                        <caesura/>
                         ata ūrdhvaṃ prakṛtipuruṣayoḥ sādharmyavaidharmye vyākhyāsyāmaḥ |
                        tad yathā --- ubhāv apy anādī, ubhāv apy anantau, ubhāv apy
                        aliṅgau, ubhāv api nityau, ubhāv apy anaparau, ubhau ca sarvagatāv
                        iti; ekā tu prakṛtir acetanā triguṇā bījadharmiṇī prasavadharmiṇy
                        amadhyasthadharmiṇī ceti, bahavas tu puruṣāś cetanāvanto 'guṇā
                        abījadharmāṇo 'prasavadharmāṇo madhyasthadharmāṇaś ceti || </l>

                    <l xml:id="SS.3.1.10">
                        <label>1931 ed. SS.3.1.10</label>
                        <caesura/>
                         tatra kāraṇānurūpaṃ kāryam iti kṛtvā sarva evaite viśeṣāḥ
                        sattvarajastamomayā bhavanti; tadañjanatvāt tanmayatvāc ca tadguṇā
                        eva puruṣā bhavantīty eke bhāṣante || </l>

                    <l xml:id="SS.3.1.11">
                        <label>1931 ed. SS.3.1.11</label>
                        <caesura/>
                         vaidyake tu ---
                        <caesura/>
                         svabhāvam īśvaraṃ kālaṃ yadṛcchāṃ niyatiṃ tathā ||
                        <caesura/>
                         pariṇāmaṃ ca manyante prakṛtiṃ pṛthudarśinaḥ || </l>

                    <l xml:id="SS.3.1.12">
                        <label>1931 ed. SS.3.1.12</label>
                        <caesura/>
                         tanmayāny eva bhūtāni tadguṇāny eva cādiśet ||
                        <caesura/>
                         taiś ca tallakṣaṇaḥ kṛtsno bhūtagrāmo vyajanyata || </l>

                    <l xml:id="SS.3.1.13">
                        <label>1931 ed. SS.3.1.13</label>
                        <caesura/>
                         tasyopayogo 'bhihitaś cikitsāṃ prati sarvadā ||
                        <caesura/>
                         bhūtebhyo hi paraṃ yasmān nāsti cintā cikitsite || </l>

                    <l xml:id="SS.3.1.14">
                        <label>1931 ed. SS.3.1.14</label>
                        <caesura/>
                         yato 'bhihitaṃ --- tatsambhavadravyasamūho bhūtādir uktaḥ;
                        bhautikāni cendriyāṇy āyurvede varṇyante, tathendriyārthāḥ || </l>

                    <l xml:id="SS.3.1.15">
                        <label>1931 ed. SS.3.1.15</label>
                        <caesura/>
                         bhavati cātra --- indriyeṇendiyārthaṃ tu svaṃ svaṃ gṛhṇāti
                        mānavaḥ || niyataṃ tulyayonitvān nānyenānyam iti sthitiḥ || </l>

                    <l xml:id="SS.3.1.16">
                        <label>1931 ed. SS.3.1.16</label>
                        <caesura/>
                         na cāyurvedaśāstreṣūpadiśyante sarvagatāḥ kṣetrajñā nityāś ca;
                        asarvagateṣu ca kṣetrajñeṣu nityapuruṣakhyāpakān hetūn udāharanti;
                        āyurvedaśāstrasiddhānteṣv asarvagatāḥ kṣetrajñā nityāś ca,
                        tiryagyonimānuṣadeveṣu saṃcaranti dharmādharmanimittaṃ; ta ete
                        'numānagrāhyāḥ paramasūkṣmāścetanāvantaḥ śāśvatā lohitaretasoḥ
                        sannipāteṣv abhivyajyante, yato 'bhihitaṃ ---
                        pañcamahābhūtaśarīrisamavāyaḥ puruṣa iti; sa eṣa(eva?)
                        karmapuruṣaś cikitsādhikṛtaḥ || </l>

                    <l xml:id="SS.3.1.17">
                        <label>1931 ed. SS.3.1.17</label>
                        <caesura/>
                         tasya sukhaduḥkhe icchādveṣau prayatnaḥ prāṇāpānāv unmeṣanimeṣau
                        buddhir manaḥ saṅkalpo vicāraṇā smṛtir vijñānam adhyavasāyo
                        viṣayopalabdhiś ca guṇāḥ || </l>

                    <l xml:id="SS.3.1.18">
                        <label>1931 ed. SS.3.1.18</label>
                        <caesura/>
                         sāttvikās tu --- ānṛśaṃsyaṃ saṃvibhāgarucitā titikṣā satyaṃ
                        dharma āstikyaṃ jñānaṃ buddhir medhā smṛtir dhṛtir anabhiṣaṅgaś
                        ca; rājasās tu --- duḥkha<pc> </pc>bahulatāṭana<pc
                        > </pc>śīlatādhṛtir ahaṅkāra ānṛtikatvam akāruṇyaṃ dambho māno
                        harṣaḥ kāmaḥ krodhaś ca; tāmasās tu --- viṣāditvaṃ nāstikyam
                        adharmaśīlatā buddher nirodho 'jñānaṃ durmedhastvam akarmaśīlatā
                        nidrālutvaṃ ceti || </l>

                    <l xml:id="SS.3.1.19">
                        <label>1931 ed. SS.3.1.19</label>
                        <caesura/>
                         āntarikṣās tu --- śabdaḥ śabdendriyaṃ sarvacchidrasamūho
                        viviktatā ca; vāyavyās tu --- sparśaḥ sparśendriyaṃ
                        sarvaceṣṭāsamūhaḥ sarvaśarīraspandanaṃ laghutā ca; taijasās tu ---
                        rūpaṃ rūpendriyaṃ varṇaḥ santāpo bhrājiṣṇutā paktir amarṣas
                        taikṣṇyaṃ śauryaṃ ca; āpyās tu --- raso rasanendriyaṃ
                        sarvadravasamūho gurutā śaityaṃ sneho retaś ca; pārthivās tu ---
                        gandho gandhendriyaṃ sarvamūrtasamūho gurutā ceti || </l>

                    <l xml:id="SS.3.1.20">
                        <label>1931 ed. SS.3.1.20</label>
                        <caesura/>
                         tatra sattvabahulam ākāśaṃ, rajobahulo vāyuḥ, sattvarajobahulo
                        'gniḥ, sattvatamobahulā āpaḥ, tamobahulā pṛthivīti || </l>



                    <l xml:id="SS.3.1.21">
                        <label>1931 ed. SS.3.1.21</label>
                        <caesura/>
                         ślokau cātra bhavataḥ |
                        <caesura/>
                         anyonyānupraviṣṭāni sarvāṇy etāni nirdiśet |
                        <caesura/>
                         sve sve dravye tu sarveṣāṃ vyaktaṃ lakṣaṇam iṣyate ||</l>



                    <l xml:id="SS.3.1.22">
